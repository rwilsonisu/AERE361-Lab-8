\section{32-bit Floating Points (Normalized and Denormalized)}
\begin{itemize}
\item \textbf{Normalized}
  \begin{itemize}
    \item Smallest positive: $1.0 \times 2^{-126} \approx 1.175 \times 10^{-38}$
    \item Largest positive: $(2 - 2^{-23}) \times 2^{127} \approx 3.403 \times 10^{38}$
    \item Smallest negative: $- (2 - 2^{-23}) \times 2^{127} \approx -3.403 \times 10^{38}$
    \item Largest negative: $-1.0 \times 2^{-126} \approx -1.175 \times 10^{-38}$
  \end{itemize}
\item \textbf{Denormalized}
  \begin{itemize}
    \item Smallest positive: $2^{-149} \approx 1.4 \times 10^{-45}$
    \item Largest positive: $(1 - 2^{-23}) \times 2^{-126} \approx 1.175 \times 10^{-38}$
    \item Smallest negative: $- (1 - 2^{-23}) \times 2^{-126} \approx -1.175 \times 10^{-38}$
    \item Largest negative: $-2^{-149} \approx -1.4 \times 10^{-45}$
  \end{itemize}
\end{itemize}
